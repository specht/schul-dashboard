def get_gradient(color, ld_mode = 'l', spread = 1.0)
    hsv = rgb_to_hsv(hex_to_rgb(color))
    hsv[0] = (hsv[0] + 360 - 60 * spread) % 360
    if ld_mode == 'l'
        hsv[1], hsv[2] = hsv[2], hsv[1]
        hsv[1] = (hsv[1] / 100.0) ** 1.5 * 100.0
    end
    left = rgb_to_hex(hsv_to_rgb(hsv))
    
    hsv = rgb_to_hsv(hex_to_rgb(color))
    hsv[0] = (hsv[0] + 90 * spread) % 360
    if ld_mode == 'l'
        hsv[1], hsv[2] = hsv[2], hsv[1]
        hsv[1] = (hsv[1] / 100.0) ** 1.5 * 100.0
    end
    right = rgb_to_hex(hsv_to_rgb(hsv))

    [left, color, right]
end

def hsv_to_rgb(c)
    h, s, v = c[0].to_f/360, c[1].to_f/100, c[2].to_f/100
    h_i = (h*6).to_i
    f = h*6 - h_i
    p = v * (1 - s)
    q = v * (1 - f*s)
    t = v * (1 - (1 - f) * s)
    r, g, b = v, t, p if h_i==0
    r, g, b = q, v, p if h_i==1
    r, g, b = p, v, t if h_i==2
    r, g, b = p, q, v if h_i==3
    r, g, b = t, p, v if h_i==4
    r, g, b = v, p, q if h_i==5
    [(r*255).to_i, (g*255).to_i, (b*255).to_i]
end

def mix(a, b, t)
    t1 = 1.0 - t
    return [a[0] * t1 + b[0] * t,
            a[1] * t1 + b[1] * t,
            a[2] * t1 + b[2] * t]
end

def rgb_to_hsv(c)
    r = c[0] / 255.0
    g = c[1] / 255.0
    b = c[2] / 255.0
    max = [r, g, b].max
    min = [r, g, b].min
    delta = max - min
    v = max * 100

    if (max != 0.0)
        s = delta / max *100
    else
        s = 0.0
    end

    if (s == 0.0)
        h = 0.0
    else
        if (r == max)
        h = (g - b) / delta
        elsif (g == max)
        h = 2 + (b - r) / delta
        elsif (b == max)
        h = 4 + (r - g) / delta
        end

        h *= 60.0

        if (h < 0)
        h += 360.0
        end
    end
    [h, s, v]
end
